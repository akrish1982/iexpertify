Free Credit Card Validator in HTML

In this tutorial, we will learn how to create a simple yet effective credit card validator using HTML, JavaScript, and CSS. This tool can help you ensure the inputted credit card number is valid before processing further transactions.

Step 1: HTML Structure


    <!DOCTYPE html>
    <html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>Credit Card Validator</title>
        <link rel="stylesheet" href="styles.css">
    </head>
    <body>
        <h1>Credit Card Validator</h1>
        <div id="card-container">
            <input type="text" id="cc-number" placeholder="Enter Credit Card Number">
            <button id="validate-btn">Validate
        </div>
        <script src="scripts.js"></script>
    </body>
    </html>

Step 3: JavaScript Validation

To validate the credit card number, we will use a simple algorithm that checks if the number is valid according to the Luhn algorithm.


    function validateCreditCard(num) {
        num = String(num);
        let sum = 0;
        for (let i = num.length - 1; i >= 0; i--) {
            const n = parseInt(num[i]);
            if (i % 2 !== 0 && n >= 5) n *= 2;
            if (n > 9) n -= 9;
            sum += n;
        }
        return sum % 10 === 0;
    }

    const validateBtn = document.getElementById('validate-btn');
    const ccNumber = document.getElementById('cc-number');

    validateBtn.addEventListener('click', function() {
        const number = ccNumber.value.trim();
        if (validateCreditCard(number)) {
            alert('The entered credit card is valid.');
        } else {
            alert('The entered credit card is invalid.');
        }
    });

Step 4: Testing

Now, simply run the HTML file in your browser, and you can test the validator using different credit card numbers. The tool will display a message to indicate whether the number is valid or invalid.
